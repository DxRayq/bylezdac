// Tablica z polskimi nazwami miesięcy const polishMonths = [ 'stycznia', 'lutego', 'marca', 'kwietnia', 'maja', 'czerwca', 'lipca', 'sierpnia', 'września', 'października', 'listopada', 'grudnia' ]; // Funkcja do konwersji daty function convertToPolishDate(dateString) { // Tworzenie obiektu Date z podanego stringu const date = new Date(dateString); // Pobieranie komponentów daty const day = date.getDate(); // Dzień const monthIndex = date.getMonth(); // Indeks miesiąca (0-11) const year = date.getFullYear(); // Rok // Pobieranie polskiej nazwy miesiąca z tablicy const polishMonth = polishMonths[monthIndex]; // Tworzenie stringu w polskim formacie return `${day} ${polishMonth} ${year}`; } // Przykład użycia const englishDate = "January 1, 2023"; const polishDate = convertToPolishDate(englishDate); console.log(polishDate); // Wynik: "1 stycznia 2023"
